# 1. 创建独立的 conda 环境，指定 Python 3.8（mmrotate 官方推荐）
conda create -n mmrotate python=3.8 -y

# 2. 激活环境
conda activate mmrotate

# 3. 安装 PyTorch 1.13.1 + CUDA 11.7（与原指令一致）
pip install torch==1.13.1+cu117 torchvision==0.14.1+cu117 --extra-index-url https://download.pytorch.org/whl/cu117

# 4. 安装 opencv-python（使用国内镜像加速）
pip install opencv-python==4.12.0.88 -i https://pypi.tuna.tsinghua.edu.cn/simple

# 5. 安装 mmcv-full 1.7.2（必须匹配 cu117 和 torch 1.13）
pip install mmcv-full==1.7.2 -f https://download.openmmlab.com/mmcv/dist/cu117/torch1.13/index.html

# 6. 安装 mmdet 2.28.2
pip install mmdet==2.28.2

# 7. 拉取 mmrotate 源码（服务器通常已安装 git）
git clone https://github.com/open-mmlab/mmrotate.git
cd mmrotate

# 8. 安装 build 依赖
pip install -r requirements/build.txt

# 9. 以可编辑模式安装 mmrotate
pip install -v -e .

# 10. 解决 yapf 版本问题（mmrotate 需要特定版本）
pip uninstall -y yapf
pip install yapf==0.40.1

# 11. 安装 send2trash（可选，用于删除文件到回收站）
pip install send2trash

# 12. 安装 DOTA_devkit（用于结果合并）
mkdir -p tools/DOTA_devkit
git clone https://github.com/CAPTAIN-WHU/DOTA_devkit.git tools/DOTA_devkit
pip install scipy shapely matplotlib opencv-python pandas
